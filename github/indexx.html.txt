            <h2>Тест сұрағы</h2>
            <p>HTML-де ең негізгі тег қандай?</p>
            <button onclick="checkAnswer('A')">A) &lt;html&gt;</button>
            <button onclick="checkAnswer('B')">B) &lt;head&gt;</button>
            <button onclick="checkAnswer('C')">C) &lt;body&gt;</button>
            <p id="result"></p>
        </section>
    </main>
    
    <script>
        function checkAnswer(answer) {
            let result = document.getElementById("result");
            if (answer === 'A') {
                result.innerHTML = "Дұрыс жауап!";
                result.style.color = "green";
            } else {
                result.innerHTML = "Қате! Дұрыс жауап: &lt;html&gt;";
                result.style.color = "red";
            }
        }
    </script>
